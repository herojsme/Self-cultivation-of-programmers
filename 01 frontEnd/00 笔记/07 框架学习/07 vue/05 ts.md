```js
  可以使用 <script> 标签上的 generic 属性声明泛型类型参数
  <script setup lang="ts" generic="T">
      defineProps<{
      items: T[]
      selected: T
      }>()
  </script>
  <script
    setup
    lang="ts"
    generic="T extends string | number, U extends Item"
  >
  import type { Item } from './types'
  defineProps<{
    id: T
    list: U[]
  }>()
  </script>
```
